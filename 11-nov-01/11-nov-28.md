### 열별 상관관계 계산 후 최대 상관계수 추출

```R
library(dplyr)

data <- palmerpenguins::penguins
data <- data %>% na.omit(data)

data %>% 
  summarise(across(everything(), ~ sum(is.na(.))))

data %>% 
  group_by(species) %>% 
  summarise(across(bill_length_mm:flipper_length_mm, 
                   ~ cor(., body_mass_g))) %>% 
  summarise(across(bill_length_mm:flipper_length_mm, 
                   ~ (.[which.max(abs(.))])))
```
